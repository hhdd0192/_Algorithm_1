팩토리얼 

```c
#include <iostream>
int factorial(int n);
int main() {
  int n;
  scanf("%d", &n);
  printf("%d", factorial(n));
}

int factorial(int n)
{
  if(n <= 1) return 1;
  return n * factorial(n-1);
}
```
