회문 구별
```c
#include <iostream>
#include <cstring>       
#include <string>

using namespace std;

int main() {
  int t;
  int count = 0;
  char str[100003];
  int len;
  
  
  scanf("%d", &t);
  for(int i = 0; i < t; i++)
    {
      int check= 0;
      scanf("%s", str);
      len = strlen(str);
      for(int j = 0; j < len/2;j++)     // 회문구별 for문
        {
          if(str[j] != str[len-j-1])
          {
            check=1;
            break;
          }
        }
      if(check == 0) count ++; // 회문 > count +1
      
    }
  printf("%d", count);  // count = 회문의 수
}
```



int 와 long long 범위 밖의 숫자 비교 
```c
#include <iostream>
#include <cstring>       
#include <string>

using namespace std;

int main() {
  string a;
  string b;
  cin >> a;
  cin >> b;

  if(a.size() > b.size())
  {
    cout << a;
  }
  else if(a.size() < b.size())
  {
    cout << b;
  }
  else
    cout << a;
}
```
