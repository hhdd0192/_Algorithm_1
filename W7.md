
대문자 -> 소문자
```c
# include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;  //cin.getline 함수 사용

int main() {
  char str[1000]; 
  cin.getline(str,sizeof(str), '\n');     //공백 포함 문자열 입력  (scanf("%[^\n]s", str); 로 대체 가능)
  for(int i = 0; i < strlen(str);i++)
    {
      if('A' <= str[i] && str[i] <= 'Z')   // 아스키 코드 값
      {
        str[i] += 32;    // 대문자 + 32 = 소문자
      }  
      printf("%c",str[i]);
    }
  }
  
```
