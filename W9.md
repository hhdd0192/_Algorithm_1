#include <iostream>
#include <utility>
#include <cstring>       
#include <string>

using namespace std;

struct student
{
int id,a,c;
string name;
};


int main() {
  int n; 
  scanf("%d", &n);
  student person[10000];

  for(int i = 0; i < n;i++)
    {
      int id,a,c;
      char name[20];
      scanf("%d %s %d %d", &id,name,&a,&c);
      person[i].id = id;
      person[i].name = name;
      person[i].a = a;
      person[i].c = c;
    }
  int m;
  scanf("%d",&m);

  for(int i = 0; i < m; i++)
    {
      int check = 1;
      char search[20];
      scanf("%s", search);
      for(int j = 0; j < n;j++)
        {
          if(person[j].name == search)
          {
            printf("%d %d %d", person[j].id,person[j].a,person[j].c);
            check = 0;
          }
          if(check == 1); printf("-1");
        }
      
    }
  
  
}

#include <iostream>
#include <utility>
#include <cstring>       
#include <string>

using namespace std;




int main() {
  int n ;
  pair <int,int> score[100000];
  scanf("%d", &n);
  for(int i=1; i <= n;i++)
    {
      int t,a;
      scanf("%d %d",&t,&a);
      score[i] = make_pair(t,a);
    }
  
  int res = 0;
  pair <int,int> best = {-1,-1};
  
  for(int j = 1; j <= n;j++)
    {
      if(score[j] > best)
      {
        best = score[j];
        res = j; 
      }
    }
  printf("%d", res);
  return 0;
}
